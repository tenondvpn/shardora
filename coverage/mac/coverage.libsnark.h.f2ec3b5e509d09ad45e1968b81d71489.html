266 template <unsigned P, unsigned M> inline FixedHash<M> bloomPart() const
267 {
268 unsigned const c_bloomBits = M * 8;
269 unsigned const c_mask = c_bloomBits - 1;
270 unsigned const c_bloomBytes = (StaticLog2<c_bloomBits>::result + 7) / 8;
271
272 static_assert((M & (M - 1)) == 0, "M must be power-of-two");
273 static_assert(P * c_bloomBytes <= N, "out of range");
274
275 FixedHash<M> ret;
276 byte const* p = data();
277 for (unsigned i = 0; i < P; ++i)
278 {
279 unsigned index = 0;
280 for (unsigned j = 0; j < c_bloomBytes; ++j, ++p)
281 index = (index << 8) | *p;
282 index &= c_mask;
283 ret[M - 1 - index / 8] |= (1 << (index % 8));
284 }
285 return ret;
286 }

303 return ret;
304 }
